bool dividecheck(const string a[], int n, int p, string divider)
{
	for (int k = 0; k < p; k++)
		if (a[k] >= divider)
			return false;
	for ( ; p < n  &&  a[p] == divider; p++)
		;
	for (int k = p; k < n; k++)
		if (a[k] <= divider)
			return false;
	string b[100];
	copy(a, a+n, b);
	sort(b, b+n);
	return equal(b, b+n, c);
}
